    <script>
        // Set today's date as default
        document.getElementById('proposalDate').value = new Date().toISOString().split('T')[0];
        
        // Auto-select product based on roof age
        document.getElementById('roofAge').addEventListener('change', function() {
            const age = parseInt(this.value);
            const productSelect = document.getElementById('gonanoProduct');
            
            if (age >= 0 && age <= 7) {
                productSelect.value = 'GoNano Shingle Saver';
            } else if (age >= 8 && age <= 15) {
                productSelect.value = 'GoNano Revive';
            } else if (age > 15) {
                productSelect.value = 'GoNano BioBoost';
            }
            
            updatePricing();
        });
        
        // Update pricing based on product selection
        document.getElementById('gonanoProduct').addEventListener('change', updatePricing);
        
        // Add event listeners for custom service fields to recalculate
        document.getElementById('service1Price').addEventListener('input', recalculateTotals);
        document.getElementById('service2Price').addEventListener('input', recalculateTotals);
        document.getElementById('service3Price').addEventListener('input', recalculateTotals);
        document.getElementById('pricePerSqFt').addEventListener('input', recalculateTotals);
        document.getElementById('squareFeet').addEventListener('input', recalculateTotals);
        document.getElementById('installationCost').addEventListener('input', recalculateTotals);
        document.getElementById('replacementCostPerSqFt').addEventListener('input', recalculateTotals);
        
        function updatePricing() {
            const product = document.getElementById('gonanoProduct').value;
            const priceField = document.getElementById('pricePerSqFt');
            
            switch(product) {
                case 'GoNano Shingle Saver':
                    priceField.value = '2.50';
                    break;
                case 'GoNano Revive':
                    priceField.value = '2.60';
                    break;
                case 'GoNano BioBoost':
                    priceField.value = '2.85';
                    break;
            }
            
            recalculateTotals();
        }
        
        function recalculateTotals() {
            const sqft = parseFloat(document.getElementById('squareFeet').value) || 0;
            const pricePerSqFt = parseFloat(document.getElementById('pricePerSqFt').value) || 0;
            const installCost = parseFloat(document.getElementById('installationCost').value) || 0;
            const replacementPerSqFt = parseFloat(document.getElementById('replacementCostPerSqFt').value) || 0;
            
            // Get custom service prices
            const service1Price = parseFloat(document.getElementById('service1Price').value) || 0;
            const service2Price = parseFloat(document.getElementById('service2Price').value) || 0;
            const service3Price = parseFloat(document.getElementById('service3Price').value) || 0;
            
            // Calculate totals
            const applicationCost = sqft * pricePerSqFt;
            const customServicesTotal = service1Price + service2Price + service3Price;
            const totalCost = applicationCost + installCost + customServicesTotal;
            const replacementCost = sqft * replacementPerSqFt;
            const savingsAmount = replacementCost - totalCost;
            
            // Update the display (if preview section exists)
            updateCostPreview({
                sqft: sqft,
                pricePerSqFt: pricePerSqFt,
                applicationCost: applicationCost.toFixed(2),
                installationCost: installCost.toFixed(2),
                service1Price: service1Price,
                service2Price: service2Price,
                service3Price: service3Price,
                customServicesTotal: customServicesTotal.toFixed(2),
                totalCost: totalCost.toFixed(2),
                replacementCost: replacementCost.toFixed(2),
                savingsAmount: savingsAmount.toFixed(2)
            });
        }
        
        function updateCostPreview(data) {
            // Create or update a preview section
            let preview = document.getElementById('costPreview');
            if (!preview) {
                preview = document.createElement('div');
                preview.id = 'costPreview';
                preview.className = 'cost-display';
                preview.style.marginTop = '20px';
                document.querySelector('form').appendChild(preview);
            }
            
            let servicesHTML = '';
            const service1Desc = document.getElementById('service1Description').value;
            const service2Desc = document.getElementById('service2Description').value;
            const service3Desc = document.getElementById('service3Description').value;
            
            if (service1Desc && data.service1Price > 0) {
                servicesHTML += `<div class="cost-row"><span>${service1Desc}:</span><strong>$${data.service1Price.toFixed(2)}</strong></div>`;
            }
            if (service2Desc && data.service2Price > 0) {
                servicesHTML += `<div class="cost-row"><span>${service2Desc}:</span><strong>$${data.service2Price.toFixed(2)}</strong></div>`;
            }
            if (service3Desc && data.service3Price > 0) {
                servicesHTML += `<div class="cost-row"><span>${service3Desc}:</span><strong>$${data.service3Price.toFixed(2)}</strong></div>`;
            }
            
            preview.innerHTML = `
                <h3 style="color: #2E8B57; margin-bottom: 15px;">Live Cost Preview</h3>
                <div class="cost-row">
                    <span>GoNano Application (${data.sqft} sq ft × $${data.pricePerSqFt}):</span>
                    <strong>$${data.applicationCost}</strong>
                </div>
                <div class="cost-row">
                    <span>Professional Installation:</span>
                    <strong>$${data.installationCost}</strong>
                </div>
                ${servicesHTML}
                <div class="cost-row total-cost">
                    <span>Total Investment:</span>
                    <strong>$${data.totalCost}</strong>
                </div>
                <div style="margin: 20px 0; padding: 15px; background-color: #fff3e0; border-radius: 5px;">
                    <div class="cost-row" style="font-size: 1.1em;">
                        <span style="color: #e65100;">Estimated Roof Replacement Cost:</span>
                        <strong style="color: #e65100;">$${data.replacementCost}</strong>
                    </div>
                </div>
                <div class="savings" style="background-color: #4CAF50; color: white; padding: 20px; border-radius: 5px; text-align: center;">
                    <div style="font-size: 1.5em;">YOUR TOTAL SAVINGS</div>
                    <div style="font-size: 2.5em; margin-top: 10px;">$${data.savingsAmount}</div>
                    <div style="font-size: 0.9em; margin-top: 5px; opacity: 0.9;">By choosing GoNano over replacement</div>
                </div>
            `;
        }
        
        // Handle form submission
        document.getElementById('proposalForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            // Get form data
            const formData = new FormData(this);
            const data = {};
            
            for (let [key, value] of formData.entries()) {
                data[key] = value;
            }
            
            // Format date
            const date = new Date(data.proposalDate);
            data.formattedDate = date.toLocaleDateString('en-US', { 
                year: 'numeric', 
                month: 'long', 
                day: 'numeric' 
            });
            
            // Calculate costs
            const sqft = parseFloat(data.squareFeet);
            const pricePerSqFt = parseFloat(data.pricePerSqFt);
            const installCost = parseFloat(data.installationCost);
            const replacementPerSqFt = parseFloat(data.replacementCostPerSqFt);
            
            // Calculate custom services total
            const service1Price = parseFloat(data.service1Price) || 0;
            const service2Price = parseFloat(data.service2Price) || 0;
            const service3Price = parseFloat(data.service3Price) || 0;
            const customServicesTotal = service1Price + service2Price + service3Price;
            
            data.applicationCost = (sqft * pricePerSqFt).toFixed(2);
            data.customServicesTotal = customServicesTotal.toFixed(2);
            data.totalCost = (parseFloat(data.applicationCost) + installCost + customServicesTotal).toFixed(2);
            data.replacementCost = (sqft * replacementPerSqFt).toFixed(2);
            data.savingsAmount = (parseFloat(data.replacementCost) - parseFloat(data.totalCost)).toFixed(2);
            
            // Display cost summary
            const costSummary = document.getElementById('costSummary');
            let customServicesHTML = '';
            
            if (data.service1Description && service1Price > 0) {
                customServicesHTML += `
                    <div class="cost-row">
                        <span>${data.service1Description}:</span>
                        <strong>$${service1Price.toFixed(2)}</strong>
                    </div>`;
            }
            if (data.service2Description && service2Price > 0) {
                customServicesHTML += `
                    <div class="cost-row">
                        <span>${data.service2Description}:</span>
                        <strong>$${service2Price.toFixed(2)}</strong>
                    </div>`;
            }
            if (data.service3Description && service3Price > 0) {
                customServicesHTML += `
                    <div class="cost-row">
                        <span>${data.service3Description}:</span>
                        <strong>$${service3Price.toFixed(2)}</strong>
                    </div>`;
            }
            
            costSummary.innerHTML = `
                <div class="cost-row">
                    <span>GoNano Application (${sqft} sq ft × $${pricePerSqFt}):</span>
                    <strong>$${data.applicationCost}</strong>
                </div>
                <div class="cost-row">
                    <span>Professional Installation:</span>
                    <strong>$${data.installationCost}</strong>
                </div>
                ${customServicesHTML}
                <div class="cost-row total-cost">
                    <span>Total Investment:</span>
                    <strong>$${data.totalCost}</strong>
                </div>
                <div style="margin: 20px 0; padding: 15px; background-color: #fff3e0; border-radius: 5px;">
                    <div class="cost-row" style="font-size: 1.1em;">
                        <span style="color: #e65100;">Estimated Roof Replacement Cost:</span>
                        <strong style="color: #e65100;">$${data.replacementCost}</strong>
                    </div>
                </div>
                <div class="savings">
                    <div style="font-size: 1.5em; color: #2E8B57;">YOUR TOTAL SAVINGS</div>
                    <div style="font-size: 2em; margin-top: 10px;">$${data.savingsAmount}</div>
                    <div style="font-size: 0.9em; margin-top: 5px; opacity: 0.8;">By choosing GoNano over replacement</div>
                </div>
            `;
            
            // Display JSON data
            document.getElementById('proposalData').textContent = JSON.stringify(data, null, 2);
            
            // Show output section
            document.getElementById('output').classList.add('active');
            
            // Scroll to output
            document.getElementById('output').scrollIntoView({ behavior: 'smooth' });
        });
        
        function copyProposalData() {
            const data = document.getElementById('proposalData').textContent;
            navigator.clipboard.writeText(data).then(function() {
                const msg = document.getElementById('successMessage');
                preview.style.display = 'block';
            setTimeout(() => msg.style.display = 'none', 3000);
            });
        }
        
        // Initialize calculations on page load
        window.addEventListener('DOMContentLoaded', function() {
            // Set default square footage if empty
            if (!document.getElementById('squareFeet').value) {
                document.getElementById('squareFeet').value = '2500';
            }
            // Trigger initial calculation
            setTimeout(recalculateTotals, 100);
        });
    </script>
